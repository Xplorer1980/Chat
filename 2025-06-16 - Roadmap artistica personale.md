Sì, puoi eliminare i **duplicati di immagini in una cartella** senza usare programmi esterni, usando **solo Python**, purché tu abbia installato qualche libreria standard come `hashlib` e `os` (già presenti). Ecco un approccio **sicuro ed efficace**:

### ✅ Metodo: confronto con **hash del contenuto**
Per ogni immagine:
1. Calcoli l’hash (MD5 o SHA256)
2. Se trovi un hash già visto, la consideri duplicata
3. Puoi stampare, spostare o eliminare i duplicati

---

### 🧠 Esempio di script (non elimina subito, prima verifica)

```python
import os
import hashlib

def calcola_hash(file_path):
    hasher = hashlib.md5()  # puoi usare anche sha256()
    with open(file_path, 'rb') as f:
        for chunk in iter(lambda: f.read(4096), b''):
            hasher.update(chunk)
    return hasher.hexdigest()

def trova_duplicati(cartella):
    hash_dict = {}
    duplicati = []

    for root, _, files in os.walk(cartella):
        for nome_file in files:
            percorso_completo = os.path.join(root, nome_file)
            try:
                file_hash = calcola_hash(percorso_completo)
                if file_hash in hash_dict:
                    duplicati.append(percorso_completo)
                else:
                    hash_dict[file_hash] = percorso_completo
            except Exception as e:
                print(f"Errore con {percorso_completo}: {e}")

    return duplicati

# 🔁 Usa questa funzione
cartella_target = "/percorso/alla/cartella"
duplicati = trova_duplicati(cartella_target)

# 🧹 Stampa o cancella (manuale per sicurezza)
for file in duplicati:
    print("Duplicato:", file)
    # os.remove(file)  # ← decommenta questa riga SOLO dopo verifica
```

### 🐍 **Script 1: Rimozione Duplicati**  
**Funzione:** Chiede una cartella, confronta i file con hash MD5, e sposta i duplicati in una sottocartella `doppioni`.

```python
import os
import hashlib
import shutil

def calcola_hash(file_path):
    hash_md5 = hashlib.md5()
    with open(file_path, "rb") as f:
        for chunk in iter(lambda: f.read(4096), b""):
            hash_md5.update(chunk)
    return hash_md5.hexdigest()

def trova_duplicati(cartella):
    visti = {}
    doppioni = []

    for root, _, files in os.walk(cartella):
        for nome_file in files:
            percorso_file = os.path.join(root, nome_file)
            hash_file = calcola_hash(percorso_file)
            if hash_file in visti:
                doppioni.append(percorso_file)
            else:
                visti[hash_file] = percorso_file

    return doppioni

def sposta_doppioni(doppioni, cartella_base):
    destinazione = os.path.join(cartella_base, "doppioni")
    os.makedirs(destinazione, exist_ok=True)
    for file in doppioni:
        shutil.move(file, os.path.join(destinazione, os.path.basename(file)))

if __name__ == "__main__":
    cartella = input("Inserisci il percorso della cartella da analizzare: ")
    doppioni = trova_duplicati(cartella)
    if doppioni:
        sposta_doppioni(doppioni, cartella)
        print(f"{len(doppioni)} file duplicati spostati nella cartella 'doppioni'.")
    else:
        print("Nessun duplicato trovato.")
```

```python
import os
import shutil
from pathlib import Path
from zipfile import ZipFile

def zip_files_in_batches(folder_path, batch_size=120):
    base_path = Path(folder_path)
    zipped_folder = base_path / "zipped"
    zip_output_folder = base_path / "ZIP"

    zipped_folder.mkdir(exist_ok=True)
    zip_output_folder.mkdir(exist_ok=True)

    all_files = sorted([f for f in base_path.iterdir() if f.is_file() and f.suffix.lower() != '.zip'])
    batch_number = 1

    while all_files:
        current_batch = all_files[:batch_size]
        zip_name = f"batch_{batch_number:03}.zip"
        zip_temp_path = base_path / zip_name
        final_zip_path = zip_output_folder / zip_name

        with ZipFile(zip_temp_path, 'w') as zipf:
            for f in current_batch:
                zipf.write(f, arcname=f.name)
                shutil.move(str(f), zipped_folder / f.name)

        shutil.move(str(zip_temp_path), final_zip_path)
        print(f"Zippati {len(current_batch)} file in: {final_zip_path}")

        batch_number += 1
        all_files = sorted([f for f in base_path.iterdir() if f.is_file() and f.suffix.lower() != '.zip'])

if __name__ == "__main__":
    folder = input("Inserisci il percorso della cartella da cui zippare i file: ").strip('"')
    zip_files_in_batches(folder)
```
