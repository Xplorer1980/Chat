```python
def installa_dipendenze(script_path):
    import importlib
    import re

    with open(script_path, "r", encoding="utf-8") as f:
        codice = f.read()

    moduli = set(re.findall(r'^\s*(?:import|from)\s+([a-zA-Z0-9_]+)', codice, re.MULTILINE))
    mancanti = []

    for modulo in moduli:
        try:
            importlib.import_module(modulo)
        except ImportError:
            mancanti.append(modulo)

    if mancanti:
        moduli_str = " ".join(mancanti)
        comando = f"pip install {moduli_str}"
        subprocess.Popen(["powershell", "-Command", comando], creationflags=subprocess.CREATE_NEW_CONSOLE)
```
