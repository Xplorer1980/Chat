## 🐍 Script Python

```python
import os

# cartella dove hai messo tutte le chat
INPUT_DIR = "./vault_chats"
OUT_TECNICO = "CACHE_SETHIS_TECNICA.md"
OUT_EMOTIVO = "CACHE_SETHIS_EMOTIVA.md"
OUT_INDICE = "INDICE_CHAT_SEMIBATCH.md"

# parole chiave banali per separare
KEYS_TECNICO = ["policy", "pipeline", "tool", "script", "errore", "cache", "indice", "python", "codice"]
KEYS_EMOTIVO = ["crisi", "entità", "burnout", "vuoto", "tradimento", "rinascita", "capitano", "senso"]

def is_tecnico(linea: str) -> bool:
    return any(k in linea.lower() for k in KEYS_TECNICO)

def is_emotivo(linea: str) -> bool:
    return any(k in linea.lower() for k in KEYS_EMOTIVO)

with open(OUT_TECNICO, "w", encoding="utf-8") as ft, \
     open(OUT_EMOTIVO, "w", encoding="utf-8") as fe, \
     open(OUT_INDICE, "w", encoding="utf-8") as fi:

    fi.write("# INDICE CHAT SEMIBATCH\n\n")

    for fname in os.listdir(INPUT_DIR):
        if not (fname.endswith(".txt") or fname.endswith(".md")):
            continue

        path = os.path.join(INPUT_DIR, fname)
        fi.write(f"- {fname}\n")

        with open(path, "r", encoding="utf-8", errors="ignore") as f:
            for linea in f:
                if is_tecnico(linea):
                    ft.write(f"[{fname}] {linea}")
                elif is_emotivo(linea):
                    fe.write(f"[{fname}] {linea}")
                else:
                    # se non rientra, lasciala in emotivo (così non si perde nulla)
                    fe.write(f"[{fname}] {linea}")
```
