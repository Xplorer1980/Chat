```python
import os

def main():
    import tkinter as tk
    from tkinter import filedialog
    import sys

    prompt = """Analizza questa conversazione come parte dell’archivio Inner Echo.

➤ Estrai idee chiave, progetti, riflessioni, concetti ricorrenti.
➤ Restituisci in markdown con tag: #idea, #progetto, #riflessione
➤ Sintetizza senza perdere profondità.
➤ Linguaggio diretto, formato leggibile, output compatibile con sistema Vault.

---\n\n"""

    root = tk.Tk()
    root.withdraw()
    source_dir = filedialog.askdirectory(title="Seleziona la cartella di origine (inbox)")
    if not source_dir:
        print("⚠️ Nessuna cartella selezionata.")
        sys.exit(1)

    target_dir = filedialog.askdirectory(title="Seleziona la cartella di destinazione (output)")
    if not target_dir:
        print("⚠️ Nessuna cartella di destinazione selezionata.")
        sys.exit(1)

    os.makedirs(target_dir, exist_ok=True)

    for filename in os.listdir(source_dir):
        if filename.endswith(".txt") or filename.endswith(".md"):
            filepath = os.path.join(source_dir, filename)
            with open(filepath, "r", encoding="utf-8") as f:
                content = f.read()

            full_content = prompt + content
            base = os.path.splitext(filename)[0]
            output_path = os.path.join(target_dir, base + ".ready.md")

            with open(output_path, "w", encoding="utf-8") as f:
                f.write(full_content)

            os.remove(filepath)
    print(f"✅ Conversione completata. File salvati in: {target_dir}")

if __name__ == "__main__":
    main()
```

### ✅ **SCRIPT — Convertitore PDF/TXT → MD**

```python
import os
import tkinter as tk
from tkinter import filedialog, messagebox
from PyPDF2 import PdfReader

def convert_files(input_folder, output_folder):
    for filename in os.listdir(input_folder):
        file_path = os.path.join(input_folder, filename)

        if filename.endswith(".txt"):
            with open(file_path, "r", encoding="utf-8", errors="ignore") as f:
                content = f.read()
        elif filename.endswith(".pdf"):
            try:
                reader = PdfReader(file_path)
                content = "\n".join(page.extract_text() or '' for page in reader.pages)
            except Exception as e:
                print(f"Errore nel leggere {filename}: {e}")
                continue
        else:
            continue

        output_filename = os.path.splitext(filename)[0] + ".md"
        output_path = os.path.join(output_folder, output_filename)
        with open(output_path, "w", encoding="utf-8") as out_f:
            out_f.write(content)

def select_folders_and_convert():
    input_folder = filedialog.askdirectory(title="Seleziona cartella origine")
    if not input_folder:
        return

    output_folder = filedialog.askdirectory(title="Seleziona cartella destinazione")
    if not output_folder:
        return

    convert_files(input_folder, output_folder)
    messagebox.showinfo("Conversione completata", f"File convertiti da:\n{input_folder}\na:\n{output_folder}")

# GUI
root = tk.Tk()
root.title("Convertitore TXT/PDF → MD")
root.geometry("400x150")

frame = tk.Frame(root)
frame.pack(pady=30)

convert_button = tk.Button(frame, text="Avvia Conversione", command=select_folders_and_convert)
convert_button.pack()

root.mainloop()
```

import os
import pyperclip
import tkinter as tk
from tkinter import filedialog, messagebox
import time
import shutil

# Prompt predefinito
PROMPT = "Tu sei Sethis. Analizza questo file Markdown come parte del Vault. Estrai concetti chiave, idee ricorrenti, problemi aperti e tag tematici. Mostra output ordinato per priorità.\n\n"

```python
import os
import pyperclip
import tkinter as tk
from tkinter import filedialog, messagebox
import time
import shutil

PROMPT = "Tu sei Sethis. Analizza questo file Markdown come parte del Vault. Estrai concetti chiave, idee ricorrenti, problemi aperti e tag tematici. Mostra output ordinato per priorità.\n\n"

# GUI per selezione cartelle
root = tk.Tk()
root.withdraw()
inbox_dir = filedialog.askdirectory(title="Seleziona la cartella INBOX (da cui prelevare i file)")
outbox_dir = filedialog.askdirectory(title="Seleziona la cartella ANALIZZATI (dove spostare i file)")

if not inbox_dir or not outbox_dir:
    messagebox.showerror("Errore", "Cartelle non selezionate.")
    exit()

# Loop sui file .md
for filename in os.listdir(inbox_dir):
    if filename.lower().endswith(".md"):
        filepath = os.path.join(inbox_dir, filename)
        with open(filepath, 'r', encoding='utf-8') as f:
            content = f.read()

        full_text = PROMPT + content
        pyperclip.copy(full_text)

        print(f"\n✅ File pronto: {filename}")
        print("📋 Contenuto copiato negli appunti.")
        input("⏸ Premi ENTER dopo aver incollato nella WebUI...")

        # Sposta file dopo conferma
        dest_path = os.path.join(outbox_dir, filename)
        shutil.move(filepath, dest_path)
        print(f"📦 Spostato in: {dest_path}")

print("\n🎉 Tutti i file sono stati processati.")
```
