### 🔧 Script: `CopiaDLL_Mancanti.ps1`

```powershell
$source = "C:\Windows\System32"
$target = "D:\Windows\System32"  # ← cambia D: con il disco vecchio

$sourceDLLs = Get-ChildItem "$source\*.dll" -Recurse
$targetDLLs = Get-ChildItem "$target\*.dll" -Recurse

$targetNames = $targetDLLs.Name

foreach ($dll in $sourceDLLs) {
    if (-not ($targetNames -contains $dll.Name)) {
        Copy-Item $dll.FullName -Destination "$target\$($dll.Name)"
    }
}
```
